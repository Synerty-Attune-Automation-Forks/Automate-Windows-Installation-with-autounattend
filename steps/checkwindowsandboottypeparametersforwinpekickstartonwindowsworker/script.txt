$values = "
    isWin10Bios = {isWin10Bios}
    isWin10Uefi = {isWin10Uefi}
    isWinServerBios = {isWinServerBios}
    isWinServerUefi = {isWinServerUefi}
    "

function printVals {
    Write-Host "${values}"
}

Write-Host "Checking ..."
printVals

[regex]$regex = 'true'

$COUNT = $regex.matches("${values}").count

Write-Host "COUNT=${COUNT}"

if ( $COUNT -ne 1 ) {
    Write-Host "FAIL: You must have exactly one only have one of the parameters above set to true"
    exit 1
}

Write-Host "SUCCESS: The parameters above are set correctly"
